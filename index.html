function loadReport(score) { fetch("/questions/report.json") .then(response => response.json()) .then(data => { const reports = data.reports; // スコアに該当するレポートを検索 const report = reports.find(r => { const [min, max] = r.range.split("-").map(Number); return score >= min && score <= max; }); showReport(report); }); } function showReport(report) { const resultDiv = document.getElementById("result"); resultDiv.innerHTML = `

${report.title}

${report.message}

おすすめ対策

    ${report.advice.map(a => `
  • ${a}
    • `).join("")}
`; }
